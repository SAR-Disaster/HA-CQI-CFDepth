/*******************************
 * CFDepth GEE 
 *
 * Input:
 *   image: uploaded 0/1 flood GeoTIFF from Google Earth Engine Imports.
 *   1 = flood and 0 = background.
 *
 * Workflow:
 *   1) Build a compact AOI from flood pixels.
 *   2) Load FABDEM and align the flood support to the DEM grid.
 *   3) Build base water surface.
 *   4) Propagate a nearest dry-edge boundary water surface through the flood support.
 *   5) Preserve valid base depths, fill unresolved pixels, and export the final CFDepth layer.
 *******************************/

var floodRaw = ee.Image(image).select(0).rename('flood_raw');

var exportName = 'CFDepth';
var exportFolder = 'FloodDepth';

var minPositiveDepth = 0.01;
var boundarySmoothRadiusPixels = 2;
var push = 5000;
var aoiBufferMeters = 1000;
var aoiDetectionScale = 30;
var depthVis = {
  min: 0,
  max: 5,
  palette: ['white', 'blue', 'purple']
};

var flood01Raw = floodRaw.gt(0).toByte().rename('flood01_raw');
var floodMaskedRaw = flood01Raw.selfMask();
var floodLonLat = ee.Image.pixelLonLat().updateMask(floodMaskedRaw);

var floodBoundsDict = floodLonLat.reduceRegion({
  reducer: ee.Reducer.minMax(),
  geometry: floodRaw.geometry(),
  scale: aoiDetectionScale,
  maxPixels: 1e13,
  bestEffort: true,
  tileScale: 4
});

var floodBounds = ee.Geometry.Rectangle([
  ee.Number(floodBoundsDict.get('longitude_min')),
  ee.Number(floodBoundsDict.get('latitude_min')),
  ee.Number(floodBoundsDict.get('longitude_max')),
  ee.Number(floodBoundsDict.get('latitude_max'))
], null, false);

var aoi = floodBounds.buffer(aoiBufferMeters).bounds();

var demCol = ee.ImageCollection('projects/sat-io/open-datasets/FABDEM')
  .filterBounds(aoi);

var firstTile = ee.Image(demCol.first()).select(0);
var demProj = firstTile.projection();
var demScale = demProj.nominalScale();

var dem = demCol
  .mosaic()
  .select(0)
  .rename('dem')
  .setDefaultProjection(demProj)
  .clip(aoi);

var floodBinary = flood01Raw
  .clip(aoi)
  .reduceResolution({
    reducer: ee.Reducer.max(),
    bestEffort: true,
    maxPixels: 4096
  })
  .reproject(demProj)
  .unmask(0)
  .toByte()
  .rename('flood_binary');

var flood = floodBinary
  .selfMask()
  .rename('flood');

function propagateNearestSurface(sourceSurface, traversalMask, maxDistance, outName) {
  var source = sourceSurface
    .mask()
    .gt(0)
    .selfMask()
    .rename('source');

  var travelCost = ee.Image.constant(val)
    .clip(aoi)
    .updateMask(traversalMask);

  var cost0 = travelCost
    .where(source, 0)
    .cumulativeCost(source, maxDistance);

  var cost1 = travelCost
    .where(source, 1)
    .cumulativeCost(source, maxDistance);

  var cost2 = sourceSurface
    .unmask(val)
    .clip(aoi)
    .updateMask(traversalMask)
    .cumulativeCost(source, maxDistance);

  var propagated = sourceSurface
    .unmask(0)
    .add(cost2.subtract(cost0).divide(cost1.subtract(cost0)))
    .rename(outName);

  return propagated.updateMask(traversalMask);
}

var outsideDem = dem.updateMask(floodBinary.eq(0));
var source = outsideDem.mask();
var val = 10000;

var cost0 = ee.Image.constant(val)
  .clip(aoi)
  .where(source, 0)
  .cumulativeCost(source, push);

var cost1 = ee.Image.constant(val)
  .clip(aoi)
  .where(source, 1)
  .cumulativeCost(source, push);

var cost2 = outsideDem
  .unmask(val)
  .clip(aoi)
  .cumulativeCost(source, push);

var costFill = cost2
  .subtract(cost0)
  .divide(cost1.subtract(cost0));

var baseSurface = outsideDem
  .unmask(0)
  .clip(aoi)
  .add(costFill)
  .rename('base_surface');

var baseDepthRaw = baseSurface
  .subtract(dem)
  .rename('base_depth_raw');

var baseDepthFlag = baseDepthRaw.unmask(-99999);

var unresolved = floodBinary.eq(1)
  .and(baseDepthFlag.lte(minPositiveDepth))
  .rename('unresolved_initial');

var outerBoundary = floodBinary
  .focal_max(1, 'square', 'pixels')
  .subtract(floodBinary)
  .gt(0)
  .and(floodBinary.eq(0))
  .selfMask()
  .rename('dry_edge_boundary');

var boundaryElev = dem
  .updateMask(outerBoundary)
  .rename('boundary_elev');

var boundarySurface = boundaryElev
  .focal_median(boundarySmoothRadiusPixels, 'square', 'pixels')
  .updateMask(outerBoundary)
  .rename('smoothed_boundary_surface');

var traversalMask = floodBinary.eq(1)
  .or(outerBoundary.unmask(0).gt(0))
  .selfMask()
  .rename('boundary_traversal_mask');

var nearestBoundaryDistance = ee.Image.constant(1)
  .clip(aoi)
  .updateMask(traversalMask)
  .cumulativeCost(boundarySurface.mask(), push)
  .rename('nearest_boundary_distance');

var nearestBoundarySurface = propagateNearestSurface(
  boundarySurface,
  traversalMask,
  push,
  'nearest_boundary_surface'
);

var boundaryConstrainedSurface = nearestBoundarySurface
  .max(dem.add(minPositiveDepth))
  .updateMask(floodBinary.eq(1))
  .rename('boundary_constrained_surface');

var surface2 = baseSurface
  .where(
    unresolved.and(boundaryConstrainedSurface.mask()),
    boundaryConstrainedSurface
  )
  .updateMask(floodBinary.eq(1))
  .max(dem.add(minPositiveDepth))
  .rename('surface_nearest_boundary_constrained');

var finalDepthRaw = surface2
  .subtract(dem)
  .rename('final_depth_raw');

var finalDepth = finalDepthRaw
  .max(0)
  .updateMask(flood)
  .rename('CFDepth');

var finalDepthDisplay = finalDepth
  .updateMask(finalDepth.gt(minPositiveDepth));

Map.setOptions('HYBRID');
Map.centerObject(aoi, 10);

Map.addLayer(
  flood,
  {palette: ['red']},
  'flood extent',
  false
);

Map.addLayer(
  outerBoundary,
  {palette: ['yellow']},
  'dry-edge boundary',
  false
);

Map.addLayer(
  boundarySurface,
  {},
  'smoothed boundary surface',
  false
);

Map.addLayer(
  nearestBoundaryDistance.updateMask(flood),
  {min: 0, max: push, palette: ['white', 'cyan', 'blue']},
  'nearest boundary distance',
  false
);

Map.addLayer(
  nearestBoundarySurface.updateMask(flood),
  {},
  'nearest boundary propagated surface',
  false
);

Map.addLayer(
  finalDepthDisplay,
  depthVis,
  'CFDepth final depth',
  true
);

Export.image.toDrive({
  image: finalDepth,
  description: exportName,
  folder: exportFolder,
  fileNamePrefix: exportName,
  region: aoi,
  scale: demScale,
  maxPixels: 1e13
});
